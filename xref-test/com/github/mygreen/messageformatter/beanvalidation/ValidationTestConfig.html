1   package com.github.mygreen.messageformatter.beanvalidation;
2   
3   import org.springframework.context.MessageSource;
4   import org.springframework.context.annotation.Bean;
5   import org.springframework.context.annotation.Configuration;
6   import org.springframework.context.annotation.Description;
7   import org.springframework.context.support.ResourceBundleMessageSource;
8   import org.springframework.validation.beanvalidation.LocalValidatorFactoryBean;
9   
10  import com.github.mygreen.messageformatter.MessageInterpolator;
11  import com.github.mygreen.messageformatter.expression.SpelExpressionEvaluator;
12  
13  @Configuration
14  public class ValidationTestConfig {
15  
16      @Bean
17      public MessageSource messageSource() {
18          final ResourceBundleMessageSource messageSource = new ResourceBundleMessageSource();
19          messageSource.addBasenames("test_messages");
20          messageSource.setDefaultEncoding("UTF-8");
21          messageSource.setUseCodeAsDefaultMessage(false);
22          messageSource.setFallbackToSystemLocale(false);
23  
24          return messageSource;
25      }
26  
27      @Description("Validator of BeanValidation")
28      @Bean
29      public LocalValidatorFactoryBean validator() {
30          final LocalValidatorFactoryBean factory = new LocalValidatorFactoryBean();
31  
32          // メッセージの処理を独自の形式に変更する。
33          CustomMessageInterpolator messageInterpolator = new CustomMessageInterpolator(
34                  messageSource(),
35                  new MessageInterpolator(new SpelExpressionEvaluator()));
36  
37          factory.setMessageInterpolator(messageInterpolator);
38  
39          return factory;
40      }
41  
42  }
