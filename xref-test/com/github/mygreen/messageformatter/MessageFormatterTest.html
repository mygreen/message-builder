17  /**
18   * {@link MessageFormatter}のテスタ
19   *
20   *
21   * @author T.TSUCHIE
22   *
23   */
24  public class MessageFormatterTest {
25  
26      private MessageFormatter messageFormatter;
27  
28      @BeforeEach
29      void setUp() throws Exception {
30          ResourceBundleMessageSource messageSource = new ResourceBundleMessageSource();
31          messageSource.addBasenames("test_messages");
32          messageSource.setDefaultEncoding("UTF-8");
33          messageSource.setFallbackToSystemLocale(false);
34  
35          MessageInterpolator messageInterpolator = new MessageInterpolator(new SpelExpressionEvaluator());
36  
37          this.messageFormatter = new MessageFormatter(messageSource, messageInterpolator);
38      }
39  
40      @Test
41      void testFormat() {
42  
43          String result = messageFormatter.create("test.message01")
44                  .param("validatedValue", 3.1)
45                  .param("min", 1)
46                  .param("max", 10)
47                  .format();
48  
49          assertThat(result).isEqualTo("メッセージ：3.10は、1～10の範囲で入力してください。");
50  
51      }
52  
53      @Test
54      void testFormatRecursively() {
55  
56          String result = messageFormatter.create("test.recursive")
57                  .param("value", "{min}")
58  //                .var("min", 3)    // min はプロパティファイルに定義
59                  .formatRecursively();
60  
61          assertThat(result).isEqualTo("再帰的なメッセージ：{abc}=3");
62  
63      }
64  
65      @Test
66      void testFormatRecursivelyMax() {
67  
68          String result = messageFormatter.create("test.recursiveMax")
69                  .param("value", "{max}")
70  //                .var("max", "{value}")    // max はプロパティファイルに定義
71                  .formatRecursively(5);
72  
73          assertThat(result).isEqualTo("再帰的なメッセージ：{abc}={max}");
74  
75      }
76  
77      @Test
78      void testVarWithAnno() {
79  
80          String result = messageFormatter.create("test.varWithAnno")
81                  .paramWithAnno("anno", Entity.class)
82                  .format();
83  
84          assertThat(result).isEqualTo("変数の初期フォーマット：アノテーション「@Entity」");
85  
86      }
87  
88      @Test
89      void testVarWithEnum() {
90  
91          String result = messageFormatter.create("test.varWithEnum")
92                  .paramWithEnum("enum", Role.Admin)
93                  .format();
94  
95          assertThat(result).isEqualTo("変数の初期フォーマット：列挙型「Role#Admin」");
96  
97      }
98  
99      @Test
100     void testVarWithClass() {
101         String result = messageFormatter.create("test.varWithClass")
102                 .paramWithClass("class", MessageFormatter.class)
103                 .format();
104 
105         assertThat(result).isEqualTo("変数の初期フォーマット：クラス「com.github.mygreen.messageformatter.MessageFormatter」");
106     }
107 
108     @Test
109     void testVarWithClasses() {
110         String result = messageFormatter.create("test.varWithClasses")
111                 .paramWithClass("classes", MessageFormatter.class, SpelExpressionEvaluator.class)
112                 .format();
113 
114         assertThat(result).isEqualTo("変数の初期フォーマット：クラス「com.github.mygreen.messageformatter.MessageFormatter, com.github.mygreen.messageformatter.expression.SpelExpressionEvaluator」");
115     }
