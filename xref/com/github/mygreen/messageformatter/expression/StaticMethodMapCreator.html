3   import java.lang.reflect.Method;
4   import java.lang.reflect.Modifier;
5   import java.util.HashMap;
6   import java.util.Map;
7   
8   import org.springframework.util.ReflectionUtils;
9   
10  import lombok.AccessLevel;
11  import lombok.NoArgsConstructor;
12  
13  /**
14   * EL式中で使用するためのstaticメソッドのマッピング情報を作成します。
15   *
16   * @author T.TSUCHIE
17   *
18   */
19  @NoArgsConstructor(access = AccessLevel.PRIVATE)
20  public class StaticMethodMapCreator {
21  
22      /**
23       * ユーティリティクラスからメソッド情報を抽出する。
24       * EL式中の独自関数として使用する。
25       * @param targetClass 抽出対象のクラス。
26       * @return メソッド名とメソッド情報のマップ。
27       */
28      public static Map<String, Method> create(final Class<?> targetClass) {
29          return create(targetClass, null);
30      }
31  
32      /**
33       * ユーティリティクラスからメソッド情報を抽出する。
34       * EL式中の独自関数として使用する。
35       * @param targetClass 抽出対象のクラス。
36       * @param prefix メソッドの接頭語(指定しない場合はnull)
37       * @return メソッド名とメソッド情報のマップ。
38       */
39      public static Map<String, Method> create(final Class<?> targetClass, final String prefix) {
40  
41          Map<String, Method> map = new HashMap<>();
42  
43          final String p = prefix == null ? "" : prefix;
44  
45          ReflectionUtils.doWithMethods(targetClass,
46                  method -> {
47                      ReflectionUtils.makeAccessible(method);
48                      map.put(p + method.getName(), method);
49                  },
50                  method -> Modifier.isPublic(method.getModifiers()) && Modifier.isStatic(method.getModifiers()));
51  
52          return map;
53  
54      }
55  }
