4   import java.util.Locale;
5   import java.util.Map;
6   import java.util.Optional;

8   import javax.validation.Path;
9   import javax.validation.metadata.ConstraintDescriptor;

11  import org.hibernate.validator.internal.engine.MessageInterpolatorContext;
12  import org.springframework.context.MessageSource;
13  import org.springframework.context.NoSuchMessageException;
14  import org.springframework.context.support.MessageSourceAccessor;
15  import org.springframework.util.StringUtils;
16  import org.springframework.validation.DefaultMessageCodesResolver;
17  import org.springframework.validation.MessageCodesResolver;
18

20  
21  import lombok.Getter;
22  import lombok.RequiredArgsConstructor;
23  import lombok.Setter;

25  /**
26   * {@link MessageInterpolator}とBeanValidationの{@link javax.validation.MessageInterpolator}をブリッジする。
27   * <p>BeanValidationのメッセージ処理をカスタマイズするために利用する。</p>
28   * <p>BeanValidationのHibernateValidatorの実装と比べ、追加で以下の変数がメッセージ中で利用が可能。</p>
29   * <ul>
30   *   <li>{@code propertyPath} : エラーとなったプロパティのパス。ネストしている場合は、{@code address.tel}のようにドットで区切られる。</li>
31   *   <li>{@code propertyName} : エラーとなったプロパティの名称。メッセージソースからプロパティ名を取得したもの。メッセージソースに定義されていなければ値は{@code null}となる。</li>
32   * </ul>
33   *
34   * @author T.TSUCHIE
35   *
36   */
37  @RequiredArgsConstructor
38  public class CustomMessageInterpolator implements javax.validation.MessageInterpolator {
39  
40      /**
41       * メッセージソース。
42       */
43      @Getter
44      private final MessageSource messageSource;
45  
46      /**
47       * パラメータ付きのメッセージのフォーマッタ。
48       */
49      @Getter
50      private final MessageInterpolator messageInterpolator;
51  
52      /**
53       * プロパティ名のメッセージコードの候補を生成する。
54       */
55      @Setter
56      @Getter
57      private MessageCodesResolver messageCodeResolver = new DefaultMessageCodesResolver();
58  
59      /**
60       * プロパティ名のメッセージコードの候補をを生成するときのコード名。
61       * デフォルトは、{@code propertyName} です。
62       * {@code null} を設定すると、コード名は付与されません。
63       */
64      @Getter
65      @Setter
66      private String propertyNameCode = "propertyName";
67  
68      @Override
69      public String interpolate(final String messageTemplate, final Context context) {
70          return messageInterpolator.interpolate(messageTemplate,
71                  createMessageVariables(context, Locale.getDefault()),
72                  0, new MessageSourceAccessor(messageSource));
73      }
74  
75      @Override
76      public String interpolate(final String messageTemplate, final Context context, final Locale locale) {
77          return messageInterpolator.interpolate(messageTemplate,
78                  createMessageVariables(context, locale),
79                  0, new MessageSourceAccessor(messageSource, locale));
80      }
81  
82      /**
83       * メッセージ中で利用可能な変数を作成する
84       * @param context コンテキスト
85       * @param locale ロケール
86       * @return メッセージ変数のマップ
87       */
88      protected Map<String, Object> createMessageVariables(final Context context, final Locale locale) {
89  
90          final Map<String, Object> vars = new HashMap<>();
91  
92          if(context instanceof MessageInterpolatorContext) {
93              MessageInterpolatorContext mic = (MessageInterpolatorContext)context;
94              vars.putAll(mic.getMessageParameters());
95  
96              // プロパティの情報を設定する
97              Path path = mic.getPropertyPath();
98              vars.put("propertyPath", path.toString());
99  
100             resolvePropertyName(mic, locale).ifPresent(name -> vars.put("propertyName", name));
101         }
102 
103         final ConstraintDescriptor<?> descriptor = context.getConstraintDescriptor();
104         for(Map.Entry<String, Object> entry : descriptor.getAttributes().entrySet()) {
105             final String attrName = entry.getKey();
106             final Object attrValue = entry.getValue();
107 
108             vars.put(attrName, attrValue);
109         }
110 
111         // 検証対象の値
112         vars.computeIfAbsent("validatedValue", key -> context.getValidatedValue());
113 
114         // デフォルトのメッセージ
115         final String defaultCode = String.format("%s.message", descriptor.getAnnotation().annotationType().getCanonicalName());
116         final String defaultMessage = messageSource.getMessage(defaultCode, null, locale);
117         if(defaultMessage == null) {
118             throw new RuntimeException(String.format("not found message code '%s'", defaultCode));
119         }
120 
121         vars.put(defaultCode, defaultMessage);
122 
123         return vars;
124 
125     }
126 
127     /**
128      * プロパティの名称をメッセージソースから解決する。
129      * <p>{@link MessageCodesResolver} で生成したコードを元に取得する。
130      * @param context コンテキスト
131      * @param locale ロケール
132      * @return プロパティの名称。解決できない場合は空を返す。
133      */
134     protected Optional<String> resolvePropertyName(final MessageInterpolatorContext context, final Locale locale) {
135 
136         final String objectName = context.getRootBeanType().getSimpleName();
137         final String field = context.getPropertyPath().toString();
138         final Class<?> fieldType = context.getValidatedValue() != null ? context.getValidatedValue().getClass() : null;
139 
140         String[] codes = messageCodeResolver.resolveMessageCodes(propertyNameCode, objectName, field, fieldType);
141 
142         for(String code : codes) {
143             try {
144                 String result = messageSource.getMessage(code, null, locale);
145                 if(StringUtils.hasLength(result)) {
146                     return Optional.of(result);
147                 }
148             } catch(NoSuchMessageException e) {
149                 /*
150                  * MessageSource#setUseCodeAsDefaultMessageの設定によって、
151                  * 対応するコードが見つからない場合、例外が発生するか戻り値がnullになったりする。
152                  */
153                 continue;
154             }
155         }
156 
157         return Optional.empty();
158     }
